version "4.5"


#include "ZScript/biribiri_base.zsc"
#include "ZScript/biribiri_events.zsc"

class LightBall : LightningActor
{
	Default
	{
		+LightningActor.DOFLASH;
		LightningActor.FlashTic 5;

		Scale 0.25;
		Translation "0:255=@80[4,8,76]";
	}
	
	States
	{
		Spawn:
			BEAM A 1 A_FadeOut(0.035);
		wait;
	}
}
class TestEmitter : Actor
{
	Default
	{
		Scale 0.25;
		+NOINTERACTION;
	}
	
	States
	{
		Spawn:
			TNT1 A 5 NoDelay
			{
				for(int i = 0; i < 360; i+=45)
				{
					BiriBiri.DoLightning(pos, (angle+frandom(0,360)+i,pitch+frandom(-90,90)), dist:200, 0.75, 10.0, 3.0);
				}
			}
		loop;
	}
}
